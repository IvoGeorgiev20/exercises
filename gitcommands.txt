git init   - създава ново локално хранилище
git clone  - изтегля дадено отдалечно хранилище
git remote -v - дава списък с адресите на всички отдалечени хранилища
git remote add origin <URL> - добавя адрес отдалечено хранилище

git add    - добавя файлове към хранилището
git commit - записва промените по файловете
git push   - изпраща промени в отдалечено хранилище

git pull   - изтегля промени от отдалечени хранилища

git status - изписва променените файлове
git diff   - показва промените между две версии на хранилището
git log    - показва дневник на всички направени промени
